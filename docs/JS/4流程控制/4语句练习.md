> prompt()可以用来获取用户输入的内容,会将输入的内容以字符串返回,通过变量接收
>
> let num = +prompt("111")
>
> alert(type of num)

## 练习1

```js
let num = parseInt(prompt("请输入一个整数"))
			//将num转换为一个整数并且通过prompt拿到输入的内容
			if(isNaN(num) || num % 1 !== 0){
				//isNaN函数判断num如果是NaN则返回true
        //如果是小数也会排除
				alert(`${num}不是一个合法数字`)
			}else{
				if(num % 2 ===0){
					alert(`${num}是偶数`)
				}else{
					alert(`${num}是奇数`)
				}
			}
```

				alert("输入正确成绩")

				if(grade ===100){
					alert("奖励一辆BMW")

				}else if (grade>=60 && grade<=79){
