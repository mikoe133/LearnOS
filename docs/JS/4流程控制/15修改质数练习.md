```js
console.time('质数练习')
for(let i= 2;i<100;i++){//外部循环从1开始遍历到99
	let confirm = true //创建一个变量记录状态，如果是质数则为TRUE则输出
	for(let j=2;j<=i ** .5;j++){//内部循环从1开始遍历到i-1；因数j只需要取到它本身的开方就行
		if(i%j===0){
			confirm = false //进入判断，取余数为0，不是质数
			break // 进入判断说明i一定不是质数，后续检查没有进行的必要
		}
	}
		
	if(confirm){
		console.log(i);//变量为TRUE，所以输出i
	}
}
//停止一个定时器
console.timeEnd('质数练习')


```
