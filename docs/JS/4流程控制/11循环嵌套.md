## 当循环发生嵌套,外层循环每执行一次,内层循环就会执行一个完整的周期

```js
		  // 打印
			// *****
			// *****
			// *****
			// *****
			// *****
      for(let i =0;i<5;i++){
				//打印的宽度
				for(let j=0;j<5;j++){
					document.write('*')
				}
				//打印的高度
				document.write('<br>')
			}
			// 打印
			// *
			// **
			// ***
			// ****
			// *****
			for(let i =0;i<5;i++){
				//打印的宽度
				for(let j=0;j<i+1;j++){
					document.write('*')
				}
				//打印的高度
				document.write('<br>')
			}
			// 打印
			// *****
			// ****
			// ***
			// **
			// *
			for(let i =0;i<5;i++){
				//打印的宽度
				for(let j=0;j<5-i;j++){
					document.write('*')
				}
				//打印的高度
				document.write('<br>')
			}
```
