##### 隐藏中间四位效果

```js
let str = "dajsdh13712323457jahsiebdhsj19293847564djsdhfgcv13811778767shudbfdsd"
//手机号规则
let re = /(1[3-9]\d)\d{4}(\d{4})/g
//开始匹配
let result
while(result = re.exec(str)){
  console.log(result[0],result[1],result[2])
  console.log(result[1]+"****"+result[2])
}
```
