#### 对象可以作为参数传递

```js
function fn(a){
	console.log(a.name)
}
let obj = {name:"薛屹阳"}
fn(obj)//薛屹阳
```

> **传递实参时,传递的不是变量本身,而是变量中存储的值**



#### 函数每次调用都会重新创建默认值

```js
        function fn(a={name:"沙和尚"}){
            console.log("a=",a);
            a.name="唐僧"
            console.log("a=",a);
        }
        fn()//沙和尚 唐僧
        fn()//沙和尚 唐僧
```

```js
        let obj = {name:"沙和尚"}
        function fn(a=obj){
            console.log("a=",a);
            a.name="唐僧"
            console.log("a=",a);
        }
        fn()//沙和尚 唐僧
        fn()//唐僧 唐僧
```
