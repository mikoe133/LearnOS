<script>
    function deleteemp() {
        //删除
        const tr = this.parentNode.parentNode
        //获取要删除的员工姓名
        const empname = tr.getElementsByTagName("td")[0].textContent
        //确认框
        let flag = confirm("确认要删除" + empname + "吗")
        if (flag) {
            tr.remove()
        }


        //事件中通过取消默认行为来阻止超链接跳转
        return false//只在xxx.xxx=function(){}这种形式绑定的事件中才适应
    }
    document.addEventListener("DOMContentLoaded", function () {
//删除
        //获取所有超链接
        const links = document.links
        for (let i = 0; i < links.length; i++) {
            links[i].onclick = deleteemp
        }
//增加
        //获取tbody
        const tbody = document.querySelector('tbody')
        //点击按钮,将用户信息插入表格
        const btn = document.getElementById('btn')
        btn.addEventListener("click", function () {
            const name = document.getElementById('name').value
            const email = document.getElementById('email').value
            const salary = document.getElementById('salary').value
            //将数据设置为DOM对象
            // tbody.insertAdjacentHTML("beforeend", `

            // </tr>
            // `)
            //创建元素
            const tr = document.createElement("tr")

            //创建td
            const nameTD =document.createElement("td")
            const emailTD =document.createElement("td")
            const salaryTD =document.createElement("td")

            emailTD.textContent = email
            salaryTD.textContent = salary

            //添加到tr中

            tr.appendChild(emailTD)
            tr.appendChild(salaryTD)
            tr.insertAdjacentHTML("beforeend","<td><a href='javascript:;'>删除</a></td>")

            tbody.appendChild(tr)
            links[links.length-1].onclick= deleteemp
            return false
        })
    })
