#### 自动切换

```js
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <style>
        .outer {
            width: 400px;
            margin: 50px auto;
            text-align: center;
        }

        img {
            width: 400px;
        }
    </style>
</head>

<body>
    <div class="outer">
        <p id="info"></p>
        <div class="img-wrapper">
            <img src="./img/2023-3-29 下午10.30拍摄的照片.jpg" alt="">
        </div>
        <div class="btn-wrapper">
            <button id="prev">-</button>
            <button id="auto">自动</button>
            <button id="next">+</button>
        </div>
    </div>
    <script>
        const info = document.getElementById('info')
        const imgArr = ["./img/2023-3-29 下午10.30拍摄的照片.jpg",
            "./img/2023-3-29 下午10.32拍摄的四联快照.jpg",
            "./img/2023-3-29 下午10.36拍摄的照片（原片）.jpg",
            "./img/2023-3-29 下午10.38拍摄的照片（原片）.jpg"]
        let current = 0//设置索引
        //自动
        const auto = document.getElementById('auto')
        let timeclose 
        auto.onclick = function () {
            //关闭定时器
            clearTimeout(timeclose)
            timeclose =setTimeout(function autotime() {
                //修改索引
                current++
                if (current > imgArr.length - 1) {
                    current = 0
                }
                img.src = imgArr[current]
                info.textContent = `总共${imgArr.length}张,当前第${current + 1}张`
                timeclose=setTimeout(autotime,100)
            }, 100)
        }
        info.textContent = `总共${imgArr.length}张,当前第${current + 1}张`
        const img = document.getElementsByTagName('img')[0]
        const prev = document.getElementById('prev')
        const next = document.getElementById('next')
        next.addEventListener("click", function () {
            clearTimeout(timeclose)
            current++
            if (current > imgArr.length - 1) {
                current = 0
            }
            img.src = imgArr[current]
            info.textContent = `总共${imgArr.length}张,当前第${current + 1}张`
        })
        prev.onclick = function () {
            clearTimeout(timeclose)
            current--
            if (current < 0) {
                current = imgArr.length - 1
            }
            img.src = imgArr[current]
            info.textContent = `总共${imgArr.length}张,当前第${current + 1}张`
        }
    </script>
</body>

</html>
```

##### 避免加载白屏

```js
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <style>
        .outer {
            position: relative;
            width: 400px;
            height: 277px;
        }

        .outer ul {
            list-style: none;
            position: relative;
            width: 100%;
            height: 100%;
        }

        .outer li {
            position: absolute;
            top: 0;
            left: 0;
            opacity: 0;
            transition: opacity .6s;
        }

        .outer li.current {
            z-index: 99;
            opacity: 1;
        }

        img {
            width: 400px;
            height: 277px;
        }

        .prev-next a {
            font-size: 60px;
            font-weight: bold;
            text-decoration: none;
            position: absolute;
            top: 0;
            bottom: 0;
            margin: auto;
            z-index: 999;
            height: 60px;
            color: white;
            opacity: .5;
        }

        .prev-next a:hover {
            opacity: 1;
        }

        #next {
            right: 0;
        }

        .dot {
            position: absolute;
            z-index: 999;
            left: 0;
            right: 0;
            display: flex;
            justify-content: center;
            bottom: 5px;

        }

        .dot a {
            margin: 5px;
            float: left;
            width: 20px;
            height: 20px;
            border-radius: 50%;
            background-color: #fff;
            opacity: .5;
        }

        .dot a:hover,
        .dot .active {
            opacity: 1;
        }
    </style>
</head>

<body>

    <div class="outer">
        <ul class="img-list">
            <li class="current"> <img src="./img/2023-3-29 下午10.30拍摄的照片.jpg" alt=""></li>
            <li><img src="./img/2023-3-29 下午10.32拍摄的四联快照.jpg" alt=""></li>
            <li><img src="./img/2023-3-29 下午10.36拍摄的照片（原片）.jpg" alt=""></li>
            <li><img src="./img/2023-3-29 下午10.38拍摄的照片（原片）.jpg" alt=""></li>
        </ul>
        <div class="prev-next">
            <a href="JavaScript:;" id="prev">
                < </a>
                    <a href="JavaScript:;" id="next"> > </a>
        </div>
        <!-- 导航点 -->
        <div class="dot">
            <a class="active" href="javascript:;"></a>
            <a href="javascript:;"></a>
            <a href="javascript:;"></a>
            <a href="javascript:;"></a>
        </div>
    </div>
    <script>
        // 获取导航点
        const dots = Array.from(document.querySelectorAll('.dot a'))
        const imgarr = Array.from(document.querySelectorAll('.img-list li'))
        document.addEventListener("click", function (event) {
            const index = dots.indexOf(event.target)
            if (index !== -1) {
                changeimg(index)
            }

        })
        const autochange = (function () {
            let timer = null
            return function () {
                if (timer === null) {
                    timer = setTimeout(function auto() {
                        changeimg("next")
                        timer = setTimeout(auto, 1000)
                    }, 1000)
                } else {
                    clearTimeout(timer)
                    timer = null
                }
            }
        })()
        autochange()
        const outer = document.getElementsByClassName('outer')[0]
        outer.onmouseenter = function () {
            autochange()
        }
        outer.onmouseleave = function () {
            autochange()
        }
        const prev = document.getElementById('prev')
        const next = document.getElementById('next')
        prev.onclick = function () {
            changeimg("prev")
        }
        next.onclick = function () {
            changeimg("next")
        }
        function changeimg(dir) {


            //获取当前显示的图片
            const current = document.querySelector('.outer .current')
            let next
            if (dir === "next") {
                next = current.nextElementSibling || imgarr[0]
            } else if (dir === "prev") {
                next = current.previousElementSibling || imgarr.at(-1)
                //切换显示状态
            } else if (typeof (dir) === "number") {
                next = imgarr[dir]

            }

            //获取要显示图片的索引
            const index = imgarr.indexOf(next)


            current.classList.remove("current")
            next.classList.add("current")
            // 点击切换导航点
            const currentactive = document.querySelector('.active ')
            currentactive.classList.remove('active')
            dots[index].classList.add('active')
        }
    </script>
</body>

</html>
```
