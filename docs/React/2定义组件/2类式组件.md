//创建类式组件
class MyComponent extends React.Component{
    render(){			//render是放在哪里的?   myComponent的原型对象上,供实例使用
      						//render中的this是谁?  myComponent的组件实例对象
        return <h2></h2>
    }
}
//渲染组件到页面
ReactDOM.render(<MyComponent/>,document.getElementById('test'))
